<script>
function validate()
{
var name = document.getElementById("nam");
var passwd = document.getElementById("pwd").value;
var phone = document.getElementById("pno").value;
if(name.value == "")
{
alert('Name cannot be blank');
return false;
}
if(name!=(/^[A-Za-z]+$/))
{
alert('Name required alphabets');
return false;
}
else if(isNaN(phone))
{
 alert('Invalid Phone Number');
 return false;
}
else if(phone.length !=10)
{
alert('Enter valid number (need 10 digits)');
return false;
}
else if(passwd == "")
{
 alert('Enter a Password');
 return false;
}
else if(passwd.length <8)
{
 alert('Password required atleast 8 characters');
 return false;
}
else if(phone == "")
{
alert('Enter your phone number!');
return false;
}
else
{
alert(' Registration Complted');
return true;
}
}
</script>
